85esp_err_t Max32664Hub::i2c_bus_init(gpio_num_t sda, gpio_num_t scl)
86{
87 // A call to i2c_bus_init should occur in sketch
88 // to avoid multiple begins with other sketches.
89
90 esp_err_t res = ESP_OK;
91
92 if (sda == GPIO_NUM_NC) {
93 sda = (gpio_num_t)I2C_MASTER_SDA;
94 }
95
96 if (scl == GPIO_NUM_NC) {
97 scl = (gpio_num_t)I2C_MASTER_SCL;
98 }
99
100#if defined(CONFIG_ENABLE_HAL_LOCKS)
101 if (_lock == nullptr) {
102 _lock = xSemaphoreCreateMutex();
103 if (_lock == nullptr) {
104 ESP_LOGE(TAG, "xSemaphoreCreateMutex failed");
105 return ESP_FAIL;
106 }
107 }
108
109 // acquire lock
110 if (xSemaphoreTake(_lock, portMAX_DELAY) != pdTRUE) {
111 ESP_LOGE(TAG, "could not acquire lock");
112 return ESP_FAIL;
113 }
114#endif
115
116 if (_i2c_started) {
117 ESP_LOGW(TAG, "Bus already started in Master Mode.");
118 return ESP_OK;
119 }
120
121 if (!_allocate_wire_buffer()) {
122 ESP_LOGE(TAG, "_allocateWireBuffer failed");
123 return ESP_FAIL;
124 }
125
126 i2c_config_t conf;
127 conf.mode = I2C_MODE_MASTER;
128 conf.sda_io_num = (int)sda;
129 conf.sda_pullup_en = GPIO_PULLUP_ENABLE;
130 conf.scl_io_num = (int)scl;
131 conf.scl_pullup_en = GPIO_PULLUP_ENABLE;
132 conf.master.clk_speed = I2C_MASTER_FREQ_HZ;
133 conf.clk_flags = I2C_SCLK_SRC_FLAG_FOR_NOMAL;
134
135 res = i2c_param_config(I2C_NUM_0, &conf);
136
137 if (res != ESP_OK) {
138 return res;
139 }
140
141 res = i2c_driver_install(I2C_NUM_0, conf.mode, 0, 0, 0);
142
143 if (res == ESP_OK) {
144 _i2c_started = true;
145 }
146
147 if (!_i2c_started) {
148 _free_wire_buffer();
149 }
150#if defined(CONFIG_ENABLE_HAL_LOCKS)
151 // release lock
152 xSemaphoreGive(_lock);
153#endif
154
155 return res;
156}

192// Family Byte: READ_DEVICE_MODE (0x02) Index Byte: 0x00, Write Byte: 0x00
193// The following function initializes the sensor. To place the MAX32664 into
194// application mode, the MFIO pin must be pulled High while the board is held
195// in reset for 10ms. After 50 addtional ms have elapsed the board should be
196// in application mode and will return two bytes, the first 0x00 is a
197// successful communcation byte, followed by 0x00 which is the byte indicating
198// which mode the IC is in.
199uint8_t Max32664Hub::begin(gpio_num_t reset_pin, gpio_num_t mfio_pin)
200{
201 if (reset_pin >= GPIO_NUM_0) {
202 _reset_pin = reset_pin;
203 gpio_set_direction(_reset_pin, GPIO_MODE_OUTPUT); // Set the pin as output
204 }
205
206 if (mfio_pin >= GPIO_NUM_0) {
207 _mfio_pin = mfio_pin;
208 gpio_set_direction(_mfio_pin, GPIO_MODE_OUTPUT); // Set the pin as output
209 }
210
211 if ((_reset_pin == GPIO_NUM_NC) || (_mfio_pin == GPIO_NUM_NC)) { // Bail if the pins have still not been defined
212 return ERR_UNKNOWN; // Return ERR_UNKNOWN
213 }
214
215 gpio_set_level(_reset_pin, GPIO_HIGH);
216 vTaskDelay(portTICK_PERIOD_MS * 30);
217
218 gpio_set_level(_mfio_pin, GPIO_HIGH);
219 gpio_set_level(_reset_pin, GPIO_LOW);
220 vTaskDelay(portTICK_PERIOD_MS * 30);
221 gpio_set_level(_reset_pin, GPIO_HIGH);
222 vTaskDelay(portTICK_PERIOD_MS * 1000);
223 gpio_set_direction(_mfio_pin, GPIO_MODE_INPUT); // To be used as an interrupt later
224 gpio_set_pull_mode(_mfio_pin, GPIO_PULLUP_ONLY); // To be used as an interrupt later
225
226 uint8_t const response_byte = _read_byte(READ_DEVICE_MODE, 0x00); // 0x00 only possible Index Byte.
227
228 return response_byte;
229}

231// Family Byte: READ_DEVICE_MODE (0x02) Index Byte: 0x00, Write Byte: 0x00
232// The following function puts the MAX32664 into bootloader mode. To place the MAX32664 into
233// bootloader mode, the MFIO pin must be pulled LOW while the board is held
234// in reset for 10ms. After 50 addtional ms have elapsed the board should be
235// in bootloader mode and will return two bytes, the first 0x00 is a
236// successful communcation byte, followed by 0x08 which is the byte indicating
237// that the board is in bootloader mode.
238uint8_t Max32664Hub::begin_bootloader(gpio_num_t reset_pin, gpio_num_t mfio_pin)
239{
240 if (reset_pin >= GPIO_NUM_0) {
241 _reset_pin = reset_pin;
242 gpio_set_direction(_reset_pin, GPIO_MODE_OUTPUT); // Set the pin as output
243 }
244
245 if (mfio_pin >= GPIO_NUM_0) {
246 _mfio_pin = mfio_pin;
247 gpio_set_direction(_mfio_pin, GPIO_MODE_OUTPUT); // Set the pin as output
248 }
249
250 if ((_reset_pin == GPIO_NUM_NC) || (_mfio_pin == GPIO_NUM_NC)) { // Bail if the pins have still not been defined
251 return ERR_UNKNOWN; // Return ERR_UNKNOWN
252 }
253
254 gpio_set_level(_mfio_pin, GPIO_LOW);
255 gpio_set_level(_reset_pin, GPIO_LOW);
256 vTaskDelay(portTICK_PERIOD_MS * 10);
257 gpio_set_level(_reset_pin, GPIO_HIGH);
258 vTaskDelay(portTICK_PERIOD_MS * 50); // Bootloader mode is enabled when this ends.
259 gpio_set_direction(_reset_pin, GPIO_MODE_OUTPUT);
260 gpio_set_direction(_mfio_pin, GPIO_MODE_OUTPUT);
261
262 // Let's check to see if the device made it into bootloader mode.
263 uint8_t const response_byte = _read_byte(READ_DEVICE_MODE, 0x00); // 0x00 only possible Index Byte
264 return response_byte;
265}

394// This function takes the 8 bytes from the FIFO buffer related to the wrist
395// heart rate algortihm: heart rate (uint16_t), confidence (uint8_t) , SpO2 (uint16_t),
396// and the finger detected status (uint8_t). Note that the the algorithm is stated as
397// "wrist" though the sensor only works with the finger. The data is loaded
398// into the whrmFifo and returned.
399BioData Max32664Hub::read_bpm()
400{
401 BioData lib_bpm;
402 uint8_t status_chauf; // The status chauffeur captures return values.
403
404 status_chauf = read_sensor_hub_status();
405
406 if (status_chauf == 1) { // Communication Error
407 lib_bpm.heart_rate = 0;
408 lib_bpm.confidence = 0;
409 lib_bpm.oxygen = 0;
410 return lib_bpm;
411 }
412
413 num_samples_out_fifo();
414
415 if (_user_selected_mode == MODE_ONE) {
416 _read_fill_array(READ_DATA_OUTPUT, READ_DATA, MAXFAST_ARRAY_SIZE, bpm_arr);
417
418 // Heart Rate formatting
419 lib_bpm.heart_rate = (uint16_t(bpm_arr[0]) << 8);
420 lib_bpm.heart_rate |= (bpm_arr[1]);
421 lib_bpm.heart_rate /= 10;
422
423 // Confidence formatting
424 lib_bpm.confidence = bpm_arr[2];
425
426 // Blood oxygen level formatting
427 lib_bpm.oxygen = uint16_t(bpm_arr[3]) << 8;
428 lib_bpm.oxygen |= bpm_arr[4];
429 lib_bpm.oxygen /= 10;
430
431 //"Machine State" - has a finger been detected?
432 lib_bpm.status = bpm_arr[5];
433
434 return lib_bpm;
435 }
436
437 else if (_user_selected_mode == MODE_TWO) {
438 _read_fill_array(READ_DATA_OUTPUT, READ_DATA, MAXFAST_ARRAY_SIZE + MAXFAST_EXTENDED_DATA, bpm_arr_two);
439
440 // Heart Rate formatting
441 lib_bpm.heart_rate = (uint16_t(bpm_arr_two[0]) << 8);
442 lib_bpm.heart_rate |= (bpm_arr_two[1]);
443 lib_bpm.heart_rate /= 10;
444
445 // Confidence formatting
446 lib_bpm.confidence = bpm_arr_two[2];
447
448 // Blood oxygen level formatting
449 lib_bpm.oxygen = uint16_t(bpm_arr_two[3]) << 8;
450 lib_bpm.oxygen |= bpm_arr_two[4];
451 lib_bpm.oxygen /= 10;
452
453 //"Machine State" - has a finger been detected?
454 lib_bpm.status = bpm_arr_two[5];
455
456 // Sp02 r Value formatting
457 uint16_t temp_val = uint16_t(bpm_arr_two[6]) << 8;
458 temp_val |= bpm_arr_two[7];
459 lib_bpm.r_value = (float)temp_val;
460 lib_bpm.r_value /= 10.0;
461
462 // Extended Machine State formatting
463 lib_bpm.ext_status = (int8_t)bpm_arr_two[8];
464
465 // There are two additional bytes of data that were requested but that
466 // have not been implemented in firmware 10.1 so will not be saved to
467 // user's data.
468 return lib_bpm;
469 }
470
471 else {
472 lib_bpm.heart_rate = 0;
473 lib_bpm.confidence = 0;
474 lib_bpm.oxygen = 0;
475 return lib_bpm;
476 }
477}

1531endTransmission() returns:
15320: success.
15331: data too long to fit in transmit buffer.
15342: received NACK on transmit of address.
15353: received NACK on transmit of data.
15364: other error.
15375: timeout
1538*/
1539uint8_t Max32664Hub::_end_transmission(bool send_stop)
1540{
1541 if (_tx_buffer == nullptr) {

1543 return 4;

1548#if defined(CONFIG_ENABLE_HAL_LOCKS)

1560 case ESP_OK:
1561 return 0;
1562 case ESP_FAIL:
