1/*
2 Original work from:
3
4 SparkFun Electronics
5 Date: 8/2019
6 Author: Elias Santistevan
7
8 This example displays a more manual method of adjusting the way in which the
9 MAX30101 gathers data. Specifically we'll look at how to modify the pulse
10 length of the LEDs within the MAX30101 which impacts the number of samples
11 that can be gathered, so we'll adjust this value as well. In addition we
12 gather additional data from the BioData type: LED samples. This data gives
13 the number of samples gathered by the MAX30101 for both the red and IR LEDs.
14 As a side note you can also choose MODE_ONE and MODE_TWO for configSensorBpm
15 as well.
16
17 If you run into an error code check the following table to help diagnose your
18 problem:
19 1 = Unavailable Command
20 2 = Unavailable Function
21 3 = Data Format Error
22 4 = Input Value Error
23 5 = Try Again
24 255 = Error Unknown
25
26 This is an example using pure ESP-IDF library.
27 OBS: Be patient, wait for the sensor be ready, keep your finger on the sensor and wait...
28 The first value is the BPM, the second is the SpO2. The values will be different from zero
29 when the sensor is sure about the values, so many zeros will be shown. If takes too long
30 try switch fingers or move your finger a little bit. Don't press too hard.
31
32 One issue that happened to me was that the sensor was not receiving proper power from the
33 3.3v pin. Try to use another power source if you get lots of timeouts.
34
35 Finally, check if `CONFIG_FREERTOS_HZ` is set to 1000 in menuconfig. If not, set it to 1000!
36
37*/

52// The familiar setup() function used on Arduino. But you can use directly in app_main()
53void setup()
54{
55 // Possible widths: 69, 118, 215, 411us
56 uint16_t width = I2C_MAX32664_PULSE_WIDTH;
57 // Possible samples: 50, 100, 200, 400, 800, 1000, 1600, 3200 samples/second
58 // Not every sample amount is possible with every width; check out our hookup
59 // guide for more information.
60 uint16_t sample_rate = I2C_MAX32664_SAMPLE_RATE;
61 uint16_t pulse_width_val;
62 uint16_t sample_val;
63
64 esp_log_level_set(TAG, ESP_LOG_VERBOSE); // Change this for less or more logging
65 esp_log_level_set("max32664", ESP_LOG_VERBOSE); // Change this for less or more logging
66
67 if (CONFIG_FREERTOS_HZ != 1000) {
68 ESP_LOGE(TAG, "CONFIG_FREERTOS_HZ must be set to 1000!");
69 esp_system_abort("CONFIG_FREERTOS_HZ must be set to 1000!");
70 }
71
72 // Initialize I2C bus before anything else
73 esp_err_t res = BIO_HUB.i2c_bus_init(I2C_MASTER_SDA, I2C_MASTER_SCL);
74 ESP_ERROR_CHECK(res);
75
76 res = (esp_err_t)BIO_HUB.begin();
77 if (res == ESP_OK) { // Zero errors!
78 ESP_LOGI(TAG, "Sensor started!");
79 }
80
81 ESP_LOGI(TAG, "Configuring Sensor....");
82 ESP_LOGI(TAG, "Keep your finger out of the sensor until you see the first readings.");
83 // Configure Sensor and BPM mode , MODE_TWO also available
84 uint8_t error = BIO_HUB.config_sensor_bpm(MODE_ONE);
85 if (error == 0) { // Zero errors.
86 ESP_LOGI(TAG, "Sensor configured.");
87 } else {
88 ESP_LOGE(TAG, "Error configuring sensor.");
89 ESP_LOGE(TAG, "Error: %u", error);
90 }
91
92 // Set pulse width.
93 error = BIO_HUB.set_pulse_width(width);
94 if (error == 0) { // Zero errors.
95 ESP_LOGI(TAG, "Pulse Width Set.");
96 } else {
97 ESP_LOGE(TAG, "Could not set Pulse Width.");
98 ESP_LOGE(TAG, "Error: %u", error);
99 }
100
101 // Check that the pulse width was set.
102 pulse_width_val = BIO_HUB.read_pulse_width();
103 ESP_LOGI(TAG, "Pulse Width: %u", pulse_width_val);
104
105 // Set sample rate per second. Remember that not every sample rate is
106 // available with every pulse width. Check hookup guide for more information.
107 error = BIO_HUB.set_sample_rate(sample_rate);
108 if (error == 0) { // Zero errors.
109 ESP_LOGI(TAG, "Sample Rate Set to %u.", sample_rate);
110 } else {
111 ESP_LOGE(TAG, "Could not set Sample Rate!");
112 ESP_LOGE(TAG, "Error: %u", error);
113 }
114
115 // BIO_HUB.set_report_period(100);
116
117 // Check sample rate.
118 sample_val = BIO_HUB.read_sample_rate();
119
120 if (sample_rate != sample_val) {
121 ESP_LOGE(TAG, "Sample Rate not set correctly!");
122 ESP_LOGE(TAG, "Sample Rate retrieved: %u", sample_val);
123 } else {
124 ESP_LOGI(TAG, "Sample rate is set to: %u", sample_val);
125 }
126
127 // Data lags a bit behind the sensor, if you're finger is on the sensor when
128 // it's being configured this delay will give some time for the data to catch
129 // up.
130 ESP_LOGI(TAG, "Loading up the buffer with data....");
131 vTaskDelay((portTICK_PERIOD_MS * 4000));
132}

134void loop()
135{
136 uint8_t samples = BIO_HUB.num_samples_out_fifo();
137 maxim::BioData body;
138
139 // Information from the readSensor function will be saved to our "body"
140 // variable.
141
142 // read all samples in fifo and use most recent one
143 while (samples) {
144 body = BIO_HUB.read_sensor_bpm();
145 --samples;
146 }
147
148 ESP_LOGI(TAG, "%u, %u", body.heart_rate, body.oxygen);
149
150 vTaskDelay((portTICK_PERIOD_MS * 10));
151}
